41  void Fill(int position) {
42  startPos = position - slopSize;
43  if (startPos + bufferSize > lenDoc)
44  startPos = lenDoc - bufferSize;
45  if (startPos < 0)
46  startPos = 0;
47  endPos = startPos + bufferSize;
48  if (endPos > lenDoc)
49  endPos = lenDoc;
50 
51  pAccess->GetCharRange(buf, startPos, endPos-startPos);
52  buf[endPos-startPos] = '\0';
53  }
